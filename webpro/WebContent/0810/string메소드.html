1. charAt/charCodeAt


str ="가나다/사자/호랑이/고양이/토끼/원숭이/곰/사자/가나다";
charAt(index) : index위치의 문자를 리턴
charCodeAt(index) : index위치의 unicode값을 리턴

2. indexOf/lastIndexOf


str ="가나다/사자/호랑이/고양이/토끼/원숭이/곰/사자/가나다";
indexOf("문자열") : 문자열의 위치를 검색
lastIndexOf("문자열") : 뒤에서부터시작하여 문자열의 위치를 검색

3. concat/trim


str ="가나다/사자/호랑이/고양이/토끼/원숭이/곰/사자/가나다";
str2 ="사과/복숭아/자두/참외/수박/바나나";
concat() : 문자열 합치기

str4 = "     사과/복숭아/자두/참외/수박/바나나    ";
trim() :앞뒤 공백제거

4. substring/slice/substr


str ="가나다/사자/호랑이/고양이/토끼/원숭이/곰/사자/가나다";

substring(start, end) : start부터 end-1까지의 글자를 추출
slice(start, end) : start부터 end-1까지의 글자를 추출
substr(start, length) : start부터 length갯수만큼 추출

5. split


str ="가나다/사자/호랑이/고양이/토끼/원숭이/곰/사자/가나다";

split(구분자, 갯수) : 구분자를 문자로 구분해서 배열로 리턴한다.

6. replace


res ="가나다/사자/호랑이/고양이/토끼/원숭이/곰/사자/가나다";

replace(old, new) : old의 값을 new의 값으로 변경한다.
replace(//g) : // -->정규식(/와 /사이에 정규표현식을 사용할 수 있다 . g는 global의 약자(old에 해당하는 전체))
